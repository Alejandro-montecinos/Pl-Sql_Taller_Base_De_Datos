CREATE TABLE CALIDAD_AIRE 
    ( 
     id_calidad NUMBER (3)  NOT NULL , 
     fecha      DATE  NOT NULL , 
     categoria  CHAR (1)  NOT NULL

;

CREATE TABLE COMUNA 
    ( 
     id_comuna                            NUMBER  NOT NULL , 
     nombre_comuna                        VARCHAR2 (1000 CHAR)  NOT NULL ,  
     id_region_provincia NUMBER (3)  NOT NULL 
    )

CREATE TABLE CONTAMINACION 
    ( 
     id_contaminacion NUMBER (3)  NOT NULL ,

CREATE TABLE CONTINENTE 
    ( 
     id_continente    NUMBER (2)  NOT NULL , 
     nombe_continente VARCHAR2 (100 CHAR)  NOT NULL 
    ) 
;

CREATE TABLE FUENTES_CONTAMINACION 
    ( 
     id_fuente                      NUMBER (3)  NOT NULL , 
     descripcion                    VARCHAR2 (100 CHAR)  NOT NULL ,

CREATE TABLE MEDICIONES 
    ( 
     id_medicion                    NUMBER (3)  NOT NULL , 
     fecha_hora                     DATE  NOT NULL , 
     instrumento_medicion           CHAR (1)  NOT NULL , 
     id_contaminacion NUMBER (3)  NOT NULL

;

CREATE TABLE PAIS 
    ( 
     id_pais                  NUMBER (3)  NOT NULL , 
     nombre_pais              VARCHAR2 (100 CHAR)  NOT NULL , 
     id_continente NUMBER (2)  NOT NULL 
    ) 
;

CREATE TABLE REGION_PROVINCIA 
    ( 
     id_region_provincia     NUMBER (3)  NOT NULL , 
     nombre_region_provincia VARCHAR2 (1000 CHAR)  NOT NULL , 
     id_pais            NUMBER (3)  NOT NULL 
    ) 
;

CREATE TABLE RESTRICCIONES

     id_restriccion   NUMBER (3)  NOT NULL ,

     fecha_inicio     DATE  NOT NULL ,
